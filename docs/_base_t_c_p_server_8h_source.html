234 int BaseTCPServer::sendBytes(SOCKET clientSocket, const DataT* const data, int size)
235 {
236 int lastSend = 0;
237 int totalSent = 0;
238
239 do
240 {
241 lastSend = send(clientSocket, reinterpret_cast<const char*>(data) + totalSent, size - totalSent, NULL);
242
243 if (lastSend == SOCKET_ERROR)
244 {
245 THROW_WEB_SERVER_EXCEPTION;
246 }
247 else if (!lastSend)
248 {
249 return totalSent;
250 }
251
252 totalSent += lastSend;
253
254 } while (totalSent < size);
255
256 return totalSent;
